**Goal:** Build 5 tax/billing UIs on top of existing CPA journal templates: VAT (ภ.พ.30) monthly submission, WHT (ภ.ง.ด.1/3/53) summary, e-Tax Invoice center, VAT Auto Journal viewer, e-Receipt Auto config.

PDF + XML export reuses existing libs (jspdf for PDF, the ขมธอ.21-2562 XML pipeline from P2-SP5 for e-Tax).

**Tech Stack:** React 18 + TypeScript + @tanstack/react-query + shadcn/ui + jspdf + exceljs + existing e-Tax XML builder

```typescript
async getVatMonthly(year: number, month: number, companyId?: string) {
  const start = new Date(Date.UTC(year, month - 1, 1));
  const end = new Date(Date.UTC(year, month, 0, 23, 59, 59));

  const lines = await this.prisma.journalLine.findMany({
    where: {
      accountCode: { in: ['21-2101', '21-2102', '11-4101', '11-2104'] },
      journalEntry: {
        postedAt: { gte: start, lte: end },
        deletedAt: null,
        ...(companyId ? { companyId } : {}),
      },
    },
    include: { journalEntry: { select: { documentNumber: true, postedAt: true, description: true, sourceType: true } } },
  });

  let vatOutput = 0;
  let vatDeferred = 0;
  let vatInput = 0;
  for (const l of lines) {
    const cr = Number(l.creditAmount);
    const dr = Number(l.debitAmount);
    if (l.accountCode === '21-2101') vatOutput += cr - dr;
    if (l.accountCode === '21-2102') vatDeferred += cr - dr;
    if (l.accountCode === '11-4101') vatInput += dr - cr;
  }

  return {
    period: { year, month, start, end },
    vatOutput,
    vatDeferred,
    vatInput,
    netVat: vatOutput - vatInput,
    lineCount: lines.length,
    lines: lines.map((l) => ({
      accountCode: l.accountCode,
      documentNumber: l.journalEntry.documentNumber,
      postedAt: l.journalEntry.postedAt,
      description: l.description ?? l.journalEntry.description,
      debit: Number(l.debitAmount),
      credit: Number(l.creditAmount),
    })),
  };
}
```

```typescript
async getWhtMonthly(year: number, month: number, companyId?: string) {
  const start = new Date(Date.UTC(year, month - 1, 1));
  const end = new Date(Date.UTC(year, month, 0, 23, 59, 59));

  // 21-3102 PND.3 ค้างจ่าย / 21-3103 PND.53 ค้างจ่าย / 21-3101 PND.1 ค้างจ่าย (payroll)
  const lines = await this.prisma.journalLine.findMany({
    where: {
      accountCode: { in: ['21-3101', '21-3102', '21-3103'] },
      journalEntry: {
        postedAt: { gte: start, lte: end },
        deletedAt: null,
        ...(companyId ? { companyId } : {}),
      },
    },
    include: { journalEntry: { select: { documentNumber: true, postedAt: true, description: true, sourceType: true, sourceId: true } } },
    orderBy: { journalEntry: { postedAt: 'asc' } },
  });

  const buckets = { PND1: { lines: [] as any[], total: 0 }, PND3: { lines: [] as any[], total: 0 }, PND53: { lines: [] as any[], total: 0 } };
  for (const l of lines) {
    const formKey = l.accountCode === '21-3101' ? 'PND1' : l.accountCode === '21-3102' ? 'PND3' : 'PND53';
    const cr = Number(l.creditAmount);
    const dr = Number(l.debitAmount);
    const amt = cr - dr; // payable accrual (Cr) minus reversals (Dr)
    buckets[formKey].lines.push({
      documentNumber: l.journalEntry.documentNumber,
      postedAt: l.journalEntry.postedAt,
      description: l.description ?? l.journalEntry.description,
      amount: amt,
    });
    buckets[formKey].total += amt;
  }

  return { period: { year, month, start, end }, ...buckets, grandTotal: buckets.PND1.total + buckets.PND3.total + buckets.PND53.total };
}
```

```typescript
async getVatAutoJournalHistory(year: number, month: number, companyId?: string) {
  const start = new Date(Date.UTC(year, month - 1, 1));

  const journalEntries = await this.prisma.journalEntry.findMany({
    where: {
      postedAt: { gte: start, lte: end },
      deletedAt: null,
      ...(companyId ? { companyId } : {}),
      lines: { some: { accountCode: { in: ['21-2101', '21-2102', '11-4101', '11-2104', '11-2105'] } } },
    },
    include: {
      lines: { where: { accountCode: { in: ['21-2101', '21-2102', '11-4101', '11-2104', '11-2105'] } } },
    },
    orderBy: { postedAt: 'desc' },
  });

  return {
    period: { year, month, start, end },
    entries: journalEntries.map((je) => ({
      id: je.id,
      documentNumber: je.documentNumber,
      postedAt: je.postedAt,
      sourceType: je.sourceType,
      description: je.description,
      vatLines: je.lines.map((l) => ({
        accountCode: l.accountCode,
        accountName: l.accountName,
        debit: Number(l.debitAmount),
        credit: Number(l.creditAmount),

}
```

```tsx
                        {d.pdfUrl && <Button variant="outline" size="sm" asChild><a href={d.pdfUrl}><Download className="size-4" /></a></Button>}
                        {d.rdStatus === 'FAILED' && <Button variant="outline" size="sm" onClick={() => resend.mutate(d.id)}><RefreshCw className="size-4" /></Button>}
                      </TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
            )}
          </QueryBoundary>
        </CardContent>
      </Card>
    </div>
  );
}

function StatusBadge({ status }: { status: string }) {
  const color = status === 'ACCEPTED' ? 'bg-emerald-500/15 text-emerald-700' : status === 'FAILED' ? 'bg-red-500/15 text-red-700' : 'bg-amber-500/15 text-amber-700';
  return <span className={`px-2 py-0.5 rounded text-xs ${color}`}>{status}</span>;
}
```

```tsx
// apps/web/src/pages/finance/EReceiptAutoPage.tsx
import { useState, useEffect } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import api from '@/lib/api';
import QueryBoundary from '@/components/QueryBoundary';
import PageHeader from '@/components/ui/PageHeader';
import { Card, CardContent, CardHeader } from '@/components/ui/card';
import { FileText, Save } from 'lucide-react';
import { Switch } from '@/components/ui/switch';
import { Button } from '@/components/ui/button';
import { toast } from 'sonner';

interface EReceiptConfig { enabled: boolean; deliveryChannel: 'LINE' | 'EMAIL' | 'BOTH'; template: 'STANDARD' | 'SIMPLE'; signerName: string; }

export default function EReceiptAutoPage() {
  const qc = useQueryClient();
  const query = useQuery({ queryKey: ['e-receipt-config'], queryFn: () => api.get<EReceiptConfig>('/system-config/e-receipt-auto').then((r) => r.data) });
  const [draft, setDraft] = useState<EReceiptConfig | null>(null);

  useEffect(() => { if (query.data && !draft) setDraft(query.data); }, [query.data, draft]);

  const save = useMutation({
    mutationFn: (cfg: EReceiptConfig) => api.put('/system-config/e-receipt-auto', cfg),
    onSuccess: () => { toast.success('บันทึกการตั้งค่าสำเร็จ'); qc.invalidateQueries({ queryKey: ['e-receipt-config'] }); },
    onError: () => toast.error('บันทึกล้มเหลว'),
  });

  return (
    <div className="space-y-6">
      <PageHeader title="ใบเสร็จอิเล็กทรอนิกส์อัตโนมัติ" icon={FileText} />
      <QueryBoundary query={query}>
        {() => draft && (
          <Card>
            <CardHeader><h3 className="font-semibold">ตั้งค่าการส่งใบเสร็จอัตโนมัติเมื่อรับชำระ</h3></CardHeader>
            <CardContent className="space-y-5">
              <div className="flex items-center justify-between">
                <div>
                  <div className="font-medium">เปิดใช้งานอัตโนมัติ</div>
                  <div className="text-sm text-muted-foreground">ส่งใบเสร็จทันทีเมื่อบันทึกการรับชำระ</div>
                </div>
                <Switch checked={draft.enabled} onCheckedChange={(v) => setDraft({ ...draft, enabled: v })} />
              </div>
              <div>
                <label className="text-sm font-medium block mb-2">ช่องทางการส่ง</label>
                <div className="flex gap-2">
                  {(['LINE', 'EMAIL', 'BOTH'] as const).map((c) => (
                    <Button key={c} variant={draft.deliveryChannel === c ? 'primary' : 'outline'} size="sm" onClick={() => setDraft({ ...draft, deliveryChannel: c })}>{c}</Button>
                  ))}
                </div>
              </div>
              <div>
                <label className="text-sm font-medium block mb-2">รูปแบบ Template</label>
                <div className="flex gap-2">
                  {(['STANDARD', 'SIMPLE'] as const).map((t) => (
                    <Button key={t} variant={draft.template === t ? 'primary' : 'outline'} size="sm" onClick={() => setDraft({ ...draft, template: t })}>{t === 'STANDARD' ? 'มาตรฐาน (เต็มรูปแบบ)' : 'แบบย่อ (สั้น)'}</Button>
                  ))}
                </div>
              </div>
              <div>
                <label className="text-sm font-medium block mb-2">ผู้เซ็นใบเสร็จ</label>
                <input className="w-full border rounded-lg p-2 bg-card" value={draft.signerName} onChange={(e) => setDraft({ ...draft, signerName: e.target.value })} />
              </div>
              <div className="flex justify-end">
                <Button onClick={() => draft && save.mutate(draft)} disabled={save.isPending}>
                  <Save className="size-4 mr-1.5" />บันทึก
                </Button>
              </div>
            </CardContent>
          </Card>
        )}
      </QueryBoundary>
    </div>
  );
}
```
